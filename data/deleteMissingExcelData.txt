Sub deleteMissingData()
Dim i As Integer
Dim count As Integer

count = 1
i = 1

Do While i < 6650 'Iterate through the percipitation data we have
    
    If (Cells(i, 20).Value = "M") Then
    
    ' If it is M, which represents missing on government of canada site
    ' store the date on the next sheet and delete the data
        
        Cells(i, 1).Copy
        Worksheets("Sheet1").Activate
        Cells(count, 1).PasteSpecial
        
        Worksheets("master - workingcopy2").Activate
        Rows(i).Delete
        
        count = count + 1
    Else
    
    i = i + 1
    
    End If
    
Loop
End

End Sub
